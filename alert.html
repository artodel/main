<script>
// Функция для отображения уведомлений
function showAlert(type = 'info', title = '', message = '', icon = '', textColor = '', bgColor = '', borderColor = '') {
    // Установки по умолчанию для разных типов
    const defaults = {
        success: {
            title: 'Успешно',
            message: 'Операция выполнена успешно',
            icon: 'fa-check-circle',
            textColor: 'text-emerald-800',
            bgColor: 'bg-emerald-50',
            borderColor: 'border-emerald-200'
        },
        error: {
            title: 'Ошибка',
            message: 'Произошла ошибка',
            icon: 'fa-times-circle',
            textColor: 'text-red-800',
            bgColor: 'bg-red-50',
            borderColor: 'border-red-200'
        },
        warning: {
            title: 'Предупреждение',
            message: 'Обратите внимание',
            icon: 'fa-exclamation-triangle',
            textColor: 'text-amber-800',
            bgColor: 'bg-amber-50',
            borderColor: 'border-amber-200'
        },
        info: {
            title: 'Информация',
            message: 'Это информационное сообщение',
            icon: 'fa-info-circle',
            textColor: 'text-blue-800',
            bgColor: 'bg-blue-50',
            borderColor: 'border-blue-200'
        }
    };

    // Используем настройки по умолчанию, если не переданы параметры
    if (defaults[type] && !title) {
        title = defaults[type].title;
        message = defaults[type].message;
        icon = defaults[type].icon;
        textColor = defaults[type].textColor;
        bgColor = defaults[type].bgColor;
        borderColor = defaults[type].borderColor;
    }

    // Создаем элемент уведомления
    const alertId = 'alert-' + Date.now();
    const alertEl = document.createElement('div');
    alertEl.id = alertId;
    alertEl.className = `alert ${bgColor} border ${borderColor} ${textColor} px-4 py-3 rounded-lg flex items-start relative shadow-md`;
    alertEl.innerHTML = `
        <i class="fas ${icon} mt-1 mr-3"></i>
        <div class="flex-1">
            <p class="font-medium">${title}</p>
            ${message ? `<p class="text-sm mt-1">${message}</p>` : ''}
        </div>
        <button onclick="closeAlert('${alertId}')" class="ml-3 text-slate-400 hover:text-slate-600">
            <i class="fas fa-times"></i>
        </button>
    `;

    // Добавляем в контейнер
    const container = document.getElementById('alert-container') || createAlertContainer();
    container.insertBefore(alertEl, container.firstChild);

    // Анимация появления
    setTimeout(() => {
        alertEl.classList.add('show');
    }, 10);

    // Автоматическое закрытие через 5 секунд (кроме ошибок)
    if (type !== 'error') {
        setTimeout(() => {
            closeAlert(alertId);
        }, 5000);
    }
}

// Закрытие уведомления
function closeAlert(id) {
    const alertEl = document.getElementById(id);
    if (alertEl) {
        alertEl.classList.remove('show');
        alertEl.classList.add('hide');
        setTimeout(() => {
            alertEl.remove();
        }, 300);
    }
}

// Создание контейнера, если его нет
function createAlertContainer() {
    const container = document.createElement('div');
    container.id = 'alert-container';
    container.className = 'fixed z-50 w-full md:w-auto md:max-w-md top-4 right-4 space-y-3';
    document.body.appendChild(container);
    return container;
}

// Инициализация контейнера при загрузке
window.addEventListener('DOMContentLoaded', () => {
    if (!document.getElementById('alert-container')) {
        createAlertContainer();
    }
});
</script>

Используйте функцию showAlert(type, title, message, icon, textColor, bgColor, borderColor) для вызова уведомлений.
